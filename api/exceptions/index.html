class GenericAction(PZPException):
    """
    Generic Action Event

    Args:
        action: action
        selected_item: selected item, if any
        ch: pressed key

    Attributes:
        action: action
        selected_item: selected item, if any
        ch: pressed key

    """

    def __init__(self, action: str, selected_item: Any, ch: str):
        super().__init__(action)
        self.action = action
        self.selected_item = selected_item
        self.ch = ch
